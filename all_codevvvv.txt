        function previewFile(input) {
            const preview  = document.getElementById('filePreview');
            const fileName = document.getElementById('fileName');
            const fileIcon = document.getElementById('fileIcon');

            if (input.files && input.files[0]) {
                const name = input.files[0].name;
                const ext  = name.split('.').pop().toLowerCase();

                const iconMap = {
                    pdf:  'fa-file-pdf',
                    doc:  'fa-file-word', docx: 'fa-file-word',
                    xls:  'fa-file-excel', xlsx: 'fa-file-excel',
                    png:  'fa-file-image', jpg: 'fa-file-image',
                    jpeg: 'fa-file-image', gif: 'fa-file-image'
                };

                fileIcon.className = 'fas ' + (iconMap[ext] || 'fa-file');
                fileName.textContent = name;
                preview.style.display = 'flex';
                document.getElementById('dropZone').style.borderColor =
                    'var(--secondary)';
            }
        }

        // Drag & drop highlight
        const zone = document.getElementById('dropZone');
        zone.addEventListener('dragover',  e => {
            e.preventDefault();
            zone.classList.add('dragover');
        });
        zone.addEventListener('dragleave', () => zone.classList.remove('dragover'));
        zone.addEventListener('drop',      e => {
            e.preventDefault();
            zone.classList.remove('dragover');
            const dt = e.dataTransfer;
            if (dt.files.length) {
                document.getElementById('fileInput').files = dt.files;
                previewFile(document.getElementById('fileInput'));
            }
        });

        function previewImage(input) {
            if (input.files && input.files[0]) {
                const reader = new FileReader();
                reader.onload = function (e) {
                    const img = document.getElementById('imgPreview');
                    img.src = e.target.result;
                    img.style.display = 'block';
                    document.getElementById('dropContent').style.display = 'none';
                    document.getElementById('dropZone').style.borderColor =
                        'var(--secondary)';
                };
                reader.readAsDataURL(input.files[0]);
            }
        }

        const zone = document.getElementById('dropZone');
        zone.addEventListener('dragover',  e => {
            e.preventDefault();
            zone.classList.add('dragover');
        });
        zone.addEventListener('dragleave', () => zone.classList.remove('dragover'));
        zone.addEventListener('drop',      e => {
            e.preventDefault();
            zone.classList.remove('dragover');
            const dt = e.dataTransfer;
            if (dt.files.length) {
                document.getElementById('fileInput').files = dt.files;
                previewImage(document.getElementById('fileInput'));
            }
        });
    </script>

    <script>
        function filterTable() {
            const search = document.getElementById('searchInput').value.toLowerCase();
            const gender = document.getElementById('genderFilter').value;
            const spec   = document.getElementById('specFilter').value;
            const rows   = document.querySelectorAll('#empTable tbody tr');

            rows.forEach(row => {
                const text      = row.innerText.toLowerCase();
                const rowGender = row.getAttribute('data-gender') || '';
                const rowSpec   = row.getAttribute('data-spec')   || '';

                const matchSearch = text.includes(search);
                const matchGender = gender === '' || rowGender === gender;
                const matchSpec   = spec   === '' || rowSpec   === spec;

                row.style.display = (matchSearch && matchGender && matchSpec)
                                    ? '' : 'none';
            });
        }
    </script>

            const chevron = document.getElementById('empChevron');

            panel.style.display        = isOpen ? 'none' : 'block';

        function updateEmpLabel() {
            const checked = document.querySelectorAll(
                'input[name="selectedEmployees"]:checked');
            const label = document.getElementById('empDropdownLabel');
            if (checked.length === 0) {
                label.textContent = '-- اختر المهندسين --';
            } else {
                label.textContent = Array.from(checked)
                    .map(cb => cb.getAttribute('data-name'))
                    .join('، ');
            }
        }

        // Close when clicking outside
        document.addEventListener('click', function (e) {
            const btn   = document.getElementById('empDropdownBtn');
            const panel = document.getElementById('empDropdownPanel');
            if (btn && panel &&
                !btn.contains(e.target) &&
                !panel.contains(e.target)) {
                panel.style.display = 'none';
                document.getElementById('empChevron')
                         .style.transform = 'rotate(0deg)';
            }
        });
